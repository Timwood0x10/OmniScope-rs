## Go 内存管理特点

1. **双重内存模型**: Go runtime GC + C manual memory
2. **defer 机制**: 延迟清理，保证资源释放
3. **finalizer**: runtime.SetFinalizer 类似析构函数
4. **cgo 指针规则**: 严格限制 Go 指针传递给 C

## 完整代码实现

```zig
const std = @import("std");
const c = @import("../ir/llvm_raw.zig").c;
const SemanticKind = @import("../semantics/semantic_tree.zig").SemanticKind;
const MemoryGraph = @import("../semantics/memory_graph.zig").MemoryGraph;
const log = @import("../common/log.zig");

pub const GoAdapter = struct {
    allocator: std.mem.Allocator,
    
    /// Go runtime 函数 (应该忽略，由GC管理)
    const GO_RUNTIME = struct {
        const ALLOC_FUNCTIONS = [_][]const u8{
            "runtime.mallocgc",
            "runtime.newobject",
            "runtime.makechan",
            "runtime.makeslice",
            "runtime.makemap",
            "runtime.convT2E",
            "runtime.convT2I",
        };
        
        const GC_FUNCTIONS = [_][]const u8{
            "runtime.gcStart",
            "runtime.gcMarkDone",
            "runtime.gcSweep",
            "runtime.GC",
        };
        
        const DEFER_FUNCTIONS = [_][]const u8{
            "runtime.deferproc",
            "runtime.deferreturn",
            "runtime.deferprocStack",
        };
        
        const FINALIZER_FUNCTIONS = [_][]const u8{
            "runtime.SetFinalizer",
        };
    };
    
    /// CGO 包装函数前缀
    const CGO_WRAPPER_PREFIX = "_Cgo_";
    const CGO_CALLBACK_PREFIX = "_cgoexp_";
    const CGO_CHECK_PREFIX = "_cgo_runtime_cgocall";
    
    pub fn init(allocator: std.mem.Allocator) GoAdapter {
        return .{ .allocator = allocator };
    }
    
    /// 检测是否是 Go runtime 函数
    pub fn isGoRuntimeFunc(func_name: []const u8) bool {
        // 检查 runtime. 前缀
        if (std.mem.startsWith(u8, func_name, "runtime.")) {
            return true;
        }
        
        // 检查已知的 runtime 函数
        for (GO_RUNTIME.ALLOC_FUNCTIONS) |f| {
            if (std.mem.eql(u8, func_name, f)) return true;
        }
        
        for (GO_RUNTIME.GC_FUNCTIONS) |f| {
            if (std.mem.eql(u8, func_name, f)) return true;
        }
        
        return false;
    }
    
    /// 检测是否是 CGO 包装函数
    pub fn isCGoWrapper(func_name: []const u8) bool {
        return std.mem.startsWith(u8, func_name, CGO_WRAPPER_PREFIX);
    }
    
    /// 检测是否是 CGO 回调
    pub fn isCGoCallback(func_name: []const u8) bool {
        return std.mem.startsWith(u8, func_name, CGO_CALLBACK_PREFIX);
    }
    
    /// 查找 defer 清理站点
    pub fn findDeferredCleanup(
        self: *GoAdapter,
        func: c.LLVMValueRef,
    ) !std.ArrayList(DeferSite) {
        var defer_sites = std.ArrayList(DeferSite).init(self.allocator);
        
        // Go 的 defer 机制在 LLVM IR 中表现为:
        // 1. runtime.deferproc 调用注册延迟函数
        // 2. runtime.deferreturn 在函数退出时执行
        
        var bb = c.LLVMGetFirstBasicBlock(func);
        while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
            var inst = c.LLVMGetFirstInstruction(bb);
            while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                if (c.LLVMIsACallInst(inst) != null) {
                    const callee = c.LLVMGetCalledValue(inst);
                    const name = std.mem.span(c.LLVMGetValueName(callee));
                    
                    if (std.mem.eql(u8, name, "runtime.deferproc")) {
                        // 找到 defer 注册点
                        // 第一个参数是延迟函数的大小
                        // 第二个参数是延迟函数的指针
                        const deferred_func = c.LLVMGetOperand(inst, 1);
                        try defer_sites.append(.{
                            .register_site = @intFromPtr(inst),
                            .deferred_func = @intFromPtr(deferred_func),
                            .semantic_kind = .go_defer_cleanup,
                        });
                        
                        log.debug("Go defer found: func=0x{x} at inst=0x{x}", .{
                            @intFromPtr(deferred_func), @intFromPtr(inst)
                        });
                    }
                }
            }
        }
        
        return defer_sites;
    }
    
    /// 检测 Go finalizer
    pub fn detectFinalizer(inst: c.LLVMValueRef) ?FinalizerInfo {
        if (c.LLVMIsACallInst(inst) == null) return null;
        
        const callee = c.LLVMGetCalledValue(inst);
        const name = std.mem.span(c.LLVMGetValueName(callee));
        
        // runtime.SetFinalizer(obj, finalizer)
        if (std.mem.eql(u8, name, "runtime.SetFinalizer")) {
            return .{
                .object = c.LLVMGetOperand(inst, 0),
                .finalizer_func = c.LLVMGetOperand(inst, 1),
                .semantic_kind = .go_finalizer,
            };
        }
        
        return null;
    }
    
    /// 分析 CGO 调用模式
    pub fn analyzeCGoCall(
        self: *GoAdapter,
        inst: c.LLVMValueRef,
    ) ?CGoCallPattern {
        _ = self;
        
        if (c.LLVMIsACallInst(inst) == null) return null;
        
        const callee = c.LLVMGetCalledValue(inst);
        const name = std.mem.span(c.LLVMGetValueName(callee));
        
        // 检测 C.malloc / C.free 模式
        if (std.mem.eql(u8, name, "C.malloc") or 
            std.mem.indexOf(u8, name, "_Cgo_malloc") != null) {
            return .{
                .pattern_type = .c_alloc,
                .c_func_name = "malloc",
                .requires_free = true,
            };
        }
        
        if (std.mem.eql(u8, name, "C.free") or 
            std.mem.indexOf(u8, name, "_Cgo_free") != null) {
            return .{
                .pattern_type = .c_free,
                .c_func_name = "free",
                .requires_free = false,
            };
        }
        
        // 检测其他 C 函数调用
        if (std.mem.startsWith(u8, name, "C.")) {
            const c_func = name[2..]; // 去掉 "C." 前缀
            return .{
                .pattern_type = .c_call,
                .c_func_name = c_func,
                .requires_free = false,
            };
        }
        
        return null;
    }
    
    /// 追踪 defer + C.free 配对
    pub fn trackDeferFreePairs(
        self: *GoAdapter,
        graph: *MemoryGraph,
        func: c.LLVMValueRef,
    ) !std.ArrayList(DeferFreePair) {
        var pairs = std.ArrayList(DeferFreePair).init(self.allocator);
        
        // 查找所有 defer 站点
        const defer_sites = try self.findDeferredCleanup(func);
        defer defer_sites.deinit();
        
        // 对于每个 defer，检查是否是 C.free
        for (defer_sites.items) |site| {
            const deferred_func_ptr = @as(*c.LLVMValueRef, @ptrFromInt(site.deferred_func));
            const deferred_func = deferred_func_ptr.*;
            
            // 检查延迟函数是否调用 C.free
            if (c.LLVMIsAFunction(deferred_func) != null) {
                var bb = c.LLVMGetFirstBasicBlock(deferred_func);
                while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
                    var inst = c.LLVMGetFirstInstruction(bb);
                    while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                        if (self.analyzeCGoCall(inst)) |pattern| {
                            if (pattern.pattern_type == .c_free) {
                                try pairs.append(.{
                                    .defer_site = site.register_site,
                                    .free_site = @intFromPtr(inst),
                                    .ptr_arg = @intFromPtr(c.LLVMGetOperand(inst, 0)),
                                });
                                
                                // 标记 MemoryGraph 中的节点有延迟清理
                                const ptr_val = @intFromPtr(c.LLVMGetOperand(inst, 0));
                                if (graph.findCanonicalAlloc(ptr_val)) |node| {
                                    node.has_deferred_cleanup = true;
                                    try node.defer_sites.append(site.register_site);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        return pairs;
    }
    
    /// 检测 Go 指针传递给 C 的违规
    pub fn detectGoPointerToC(
        self: *GoAdapter,
        inst: c.LLVMValueRef,
    ) ?GoPointerViolation {
        _ = self;
        
        // CGO 规则: 不能将指向 Go 内存的指针传递给 C
        // 除非该内存不包含指针
        
        if (c.LLVMIsACallInst(inst) == null) return null;
        
        const callee = c.LLVMGetCalledValue(inst);
        const name = std.mem.span(c.LLVMGetValueName(callee));
        
        // 如果是 C 函数调用
        if (std.mem.startsWith(u8, name, "C.") or isCGoWrapper(name)) {
            // 检查参数是否来自 Go runtime 分配
            const num_args = c.LLVMGetNumArgOperands(inst);
            var i: u32 = 0;
            while (i < num_args) : (i += 1) {
                const arg = c.LLVMGetOperand(inst, i);
                
                // 检查是否来自 runtime.mallocgc 等
                // TODO: 实现完整的指针来源追踪
                _ = arg;
            }
        }
        
        return null;
    }
};

pub const DeferSite = struct {
    register_site: u64,      // runtime.deferproc 调用位置
    deferred_func: u64,      // 延迟函数指针
    semantic_kind: SemanticKind,
};

pub const FinalizerInfo = struct {
    object: c.LLVMValueRef,
    finalizer_func: c.LLVMValueRef,
    semantic_kind: SemanticKind,
};

pub const CGoCallPattern = struct {
    pattern_type: enum {
        c_alloc,   // C.malloc
        c_free,    // C.free
        c_call,    // 其他 C 函数
    },
    c_func_name: []const u8,
    requires_free: bool,
};

pub const DeferFreePair = struct {
    defer_site: u64,
    free_site: u64,
    ptr_arg: u64,
};

pub const GoPointerViolation = struct {
    call_site: u64,
    go_ptr: u64,
    c_func_name: []const u8,
};
```

## 🔬 TinyGo Adapter

### 文件位置: `src/lang/tinygo_adapter.zig` (新建)

TinyGo 是 Go 的子集，主要用于嵌入式和 WASM。

```zig
const std = @import("std");
const GoAdapter = @import("go_adapter.zig").GoAdapter;

pub const TinyGoAdapter = struct {
    go_adapter: GoAdapter,
    
    /// TinyGo runtime 函数 (更少，更简单)
    const TINYGO_RUNTIME = struct {
        const ALLOC_FUNCTIONS = [_][]const u8{
            "runtime.alloc",
            "runtime.free",
            "runtime.realloc",
        };
    };
    
    pub fn init(allocator: std.mem.Allocator) TinyGoAdapter {
        return .{
            .go_adapter = GoAdapter.init(allocator),
        };
    }
    
    /// TinyGo 没有完整的 GC，更接近手动内存管理
    pub fn isTinyGoRuntimeFunc(func_name: []const u8) bool {
        for (TINYGO_RUNTIME.ALLOC_FUNCTIONS) |f| {
            if (std.mem.eql(u8, func_name, f)) return true;
        }
        return false;
    }
    
    /// TinyGo 的 defer 实现更简单
    pub fn findDeferredCleanup(
        self: *TinyGoAdapter,
        func: c.LLVMValueRef,
    ) !std.ArrayList(GoAdapter.DeferSite) {
        // 复用 Go adapter 的逻辑
        return self.go_adapter.findDeferredCleanup(func);
    }
};
```

## 使用示例

```zig
// 在 PassManager 中集成 Go Adapter
const go_adapter = try GoAdapter.init(allocator);

// 查找 defer 清理
const defer_sites = try go_adapter.findDeferredCleanup(func);
defer defer_sites.deinit();

log.info("Found {d} defer sites in function", .{defer_sites.items.len});

// 追踪 defer + C.free 配对
const pairs = try go_adapter.trackDeferFreePairs(&memory_graph, func);
defer pairs.deinit();

for (pairs.items) |pair| {
    log.info("defer C.free pair: defer=0x{x} free=0x{x}", .{
        pair.defer_site, pair.free_site
    });
}
```
