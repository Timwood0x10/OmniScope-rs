### 2.2 回归测试套件

**文件**: `tests/regression/regression_suite.zig`

```zig
const RegressionTest = struct {
    name: []const u8,
    bc_file: []const u8,
    expected_issues: u32,
    expected_critical: u32,
    expected_high: u32,
    max_fp_rate: f32,
};

const REGRESSION_TESTS = [_]RegressionTest{
    .{
        .name = "wasmtime",
        .bc_file = "corpus/wasmtime_test.bc",
        .expected_issues = 567,
        .expected_critical = 1,
        .expected_high = 566,
        .max_fp_rate = 0.05,  // 5% FP 容忍度
    },
    .{
        .name = "bun_alloc",
        .bc_file = "corpus/real_project_test/bun_alloc.bc",
        .expected_issues = 1,  // 只有1个真实问题
        .expected_critical = 0,
        .expected_high = 0,
        .max_fp_rate = 0.0,  // 0% FP (完美精度)
    },
    .{
        .name = "bun_jsc",
        .bc_file = "corpus/real_project_test/bun_core.bc",
        .expected_issues = 5,  // 消除FP后
        .expected_critical = 1,
        .expected_high = 2,
        .max_fp_rate = 0.4,  // 40% FP 容忍度
    },
};

test "Regression: run all regression tests" {
    for (REGRESSION_TESTS) |test_case| {
        std.debug.print("\n=== Running regression test: {s} ===\n", .{test_case.name});
        
        const result = try runAnalysis(test_case.bc_file);
        defer result.deinit();
        
        // 验证 issue 数量
        try testing.expectApproxEqAbs(
            @as(f32, @floatFromInt(test_case.expected_issues)),
            @as(f32, @floatFromInt(result.total_issues)),
            @as(f32, @floatFromInt(test_case.expected_issues)) * 0.1,  // 10% 容忍度
        );
        
        // 验证严重度分布
        try testing.expectEqual(test_case.expected_critical, result.critical_count);
        
        // 验证 FP 率
        const fp_rate = calculateFPRate(result);
        try testing.expect(fp_rate <= test_case.max_fp_rate);
        
        std.debug.print("✅ {s}: {d} issues, FP rate: {d:.2}%\n", .{
            test_case.name,
            result.total_issues,
            fp_rate * 100,
        });
    }
}
```

```zig
const BenchmarkCase = struct {

        .file_size_mb = 0.1,
        .max_time_ms = 200,
        .max_memory_mb = 50,
    },
    .{
        .name = "medium (100-500 funcs)",
        .bc_file = "corpus/medium_test.bc",
        .file_size_mb = 0.5,
        .max_time_ms = 1000,
        .max_memory_mb = 100,
    },
    .{
        .name = "large (500-3000 funcs)",
        .bc_file = "corpus/sqlite3.bc",
        .file_size_mb = 5.0,
        .max_time_ms = 15000,
        .max_memory_mb = 500,
    },
    .{
        .name = "huge (>3000 funcs)",
        .bc_file = "corpus/large_project.bc",
        .file_size_mb = 10.0,
        .max_time_ms = 30000,
        .max_memory_mb = 1000,
    },
};

test "Performance: benchmark all sizes" {
    for (BENCHMARKS) |bench| {
        std.debug.print("\n=== Benchmark: {s} ({d:.1} MB) ===\n", .{
            bench.name,
            bench.file_size_mb,
        });
        
        const start_time = std.time.milliTimestamp();
        const start_mem = getCurrentMemoryUsage();
        
        const result = try runAnalysis(bench.bc_file);
        defer result.deinit();
        
        const elapsed_ms = std.time.milliTimestamp() - start_time;
        const peak_mem_mb = (getPeakMemoryUsage() - start_mem) / (1024 * 1024);
        
        std.debug.print("  Time: {d} ms (max: {d} ms)\n", .{ elapsed_ms, bench.max_time_ms });
        std.debug.print("  Memory: {d} MB (max: {d} MB)\n", .{ peak_mem_mb, bench.max_memory_mb });
        
        // 验证性能指标
        try testing.expect(elapsed_ms <= bench.max_time_ms);
        try testing.expect(peak_mem_mb <= bench.max_memory_mb);
    }
}
```

### 4.2 自动化测试脚本

**文件**: `scripts/run_full_validation.sh`

```bash
#!/bin/bash
set -e

OMNISCOPE="./zig-out/bin/OmniScope"
RESULTS_DIR="validation_results"
mkdir -p "$RESULTS_DIR"

echo "=== OmniScope 0.2.0 Full Validation Suite ==="
echo ""

# 测试计数器
TOTAL=0
PASSED=0
FAILED=0

# 测试函数
run_test() {
    local name=$1
    local bc_file=$2
    local expected_issues=$3
    local max_fp_rate=$4
    
    TOTAL=$((TOTAL + 1))
    echo "[$TOTAL] Testing $name..."
    
    # 运行分析
    if $OMNISCOPE "$bc_file" --json > "$RESULTS_DIR/$name.json" 2>&1; then
        # 解析结果
        actual_issues=$(jq '.summary.total_issues' "$RESULTS_DIR/$name.json")
        
        # 计算差异
        diff=$((actual_issues - expected_issues))
        diff_pct=$(echo "scale=2; $diff * 100 / $expected_issues" | bc)
        
        if (( $(echo "$diff_pct < 10" | bc -l) )); then
            echo "  ✅ PASS: $actual_issues issues (expected ~$expected_issues)"
            PASSED=$((PASSED + 1))
        else
            echo "  ❌ FAIL: $actual_issues issues (expected ~$expected_issues, diff: $diff_pct%)"
            FAILED=$((FAILED + 1))
        fi
    else
        echo "  ❌ FAIL: Analysis crashed"
        FAILED=$((FAILED + 1))
    fi
}

# 运行所有测试
run_test "wasmtime" "corpus/wasmtime_test.bc" 567 0.05
run_test "bun_alloc" "corpus/real_project_test/bun_alloc.bc" 1 0.0
run_test "bun_jsc" "corpus/real_project_test/bun_core.bc" 5 0.4
run_test "ring" "corpus/ring.bc" 16 0.1

# 总结
echo ""
echo "=== Summary ==="
echo "Total: $TOTAL"
echo "Passed: $PASSED"
echo "Failed: $FAILED"
echo "Success Rate: $(echo "scale=2; $PASSED * 100 / $TOTAL" | bc)%"

exit $FAILED
```
