# OmniScope 0.2.0 详细开发方案 - Part 2: FFI契约数据库

## 🗄️ FFI契约数据库设计

### 核心概念

**FFI契约** = 一个FFI函数的内存管理语义的完整描述

**为什么需要契约数据库？**
1. **消除误报**: 识别GC管理的对象（如JNI LocalRef），不报告为泄漏
2. **发现真实bug**: 检测错误的配对（如SSL_CTX_new + BIO_free）
3. **提高精度**: 基于已知的库行为做出准确判断

```zig
const std = @import("std");

/// FFI契约 - 描述一个FFI函数的内存管理语义
pub const FFIContract = struct {
    /// 函数名 (如 "SSL_CTX_new", "PyBytes_FromString")
    function_name: []const u8,
    
    /// 契约类型
    contract_type: ContractType,
    
    /// 配对的释放函数 (如果是allocator)
    paired_release: ?[]const []const u8 = null,
    
    /// 所有权语义
    ownership: OwnershipSemantics,
    
    /// 是否容易出错 (已知的bug高发点)
    error_prone: bool = false,
    
    /// 语言/库来源
    source: ContractSource,
    
    /// 可选的文档链接
    documentation_url: ?[]const u8 = null,
};

pub const ContractType = enum {
    allocator,        // 分配资源 (malloc, SSL_CTX_new)
    deallocator,      // 释放资源 (free, SSL_CTX_free)
    borrower,         // 借用引用 (PyList_GetItem)
    transfer,         // 所有权转移 (Box::into_raw)
    refcount_inc,     // 增加引用计数 (Py_INCREF)
    refcount_dec,     // 减少引用计数 (Py_DECREF)
    gc_managed,       // GC管理 (Java new, Go make)
};

pub const OwnershipSemantics = enum {
    caller_owns,      // 调用者拥有返回值，必须释放
    callee_owns,      // 被调用者拥有，调用者不应释放
    borrowed,         // 借用，不转移所有权
    transferred,      // 所有权转移给调用者
    shared,           // 共享所有权 (引用计数)
};

pub const ContractSource = enum {
    openssl,
    boringssl,
    sqlite,
    python_capi,
    jni,
    posix,
    glib,
    javascriptcore,
    libuv,
    curl,
    zlib,
    custom,
};
```

```zig
/// FFI契约数据库
pub const FFIContractDB = struct {
    contracts: std.StringHashMap(FFIContract),
    allocator: std.mem.Allocator,
    
    pub fn init(allocator: std.mem.Allocator) !FFIContractDB {
        var db = FFIContractDB{
            .contracts = std.StringHashMap(FFIContract).init(allocator),
            .allocator = allocator,
        };
        
        // 加载内置契约
        try db.loadBuiltinContracts();
        
        return db;
    }
    
    pub fn deinit(self: *FFIContractDB) void {
        self.contracts.deinit();
    }
    
    /// 加载内置契约规则
    fn loadBuiltinContracts(self: *FFIContractDB) !void {
        // ========== OpenSSL/BoringSSL ==========
        try self.addContract(.{
            .function_name = "SSL_CTX_new",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"SSL_CTX_free"},
            .ownership = .caller_owns,
            .error_prone = true,
            .source = .openssl,
            .documentation_url = "https://www.openssl.org/docs/man1.1.1/man3/SSL_CTX_new.html",
        });
        
        try self.addContract(.{
            .function_name = "SSL_CTX_free",
            .contract_type = .deallocator,
            .ownership = .transferred,
            .source = .openssl,
        });
        
        try self.addContract(.{
            .function_name = "SSL_new",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"SSL_free"},
            .ownership = .caller_owns,
            .error_prone = true,
            .source = .openssl,
        });
        
        try self.addContract(.{
            .function_name = "BIO_new",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"BIO_free", "BIO_free_all"},
            .ownership = .caller_owns,
            .source = .openssl,
        });
        
        // ========== Python C API ==========
        try self.addContract(.{
            .function_name = "PyBytes_FromString",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"Py_DECREF", "Py_XDECREF"},
            .ownership = .caller_owns,
            .error_prone = false,
            .source = .python_capi,
        });
        
        try self.addContract(.{
            .function_name = "PyList_GetItem",
            .contract_type = .borrower,
            .ownership = .borrowed,
            .error_prone = false,
            .source = .python_capi,
        });
        
        try self.addContract(.{
            .function_name = "PyList_New",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"Py_DECREF"},
            .ownership = .caller_owns,
            .source = .python_capi,
        });
        
        try self.addContract(.{
            .function_name = "Py_INCREF",
            .contract_type = .refcount_inc,
            .ownership = .shared,
            .source = .python_capi,
        });
        
        try self.addContract(.{
            .function_name = "Py_DECREF",
            .contract_type = .refcount_dec,
            .ownership = .transferred,
            .source = .python_capi,
        });
        
        // ========== SQLite ==========
        try self.addContract(.{
            .function_name = "sqlite3_open",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"sqlite3_close"},
            .ownership = .caller_owns,
            .error_prone = true,
            .source = .sqlite,
        });
        
        try self.addContract(.{
            .function_name = "sqlite3_prepare_v2",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"sqlite3_finalize"},
            .ownership = .caller_owns,
            .error_prone = true,
            .source = .sqlite,
        });
        
        // ========== JNI (Java Native Interface) ==========
        try self.addContract(.{
            .function_name = "NewGlobalRef",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"DeleteGlobalRef"},
            .ownership = .caller_owns,
            .error_prone = true,
            .source = .jni,
        });
        
        try self.addContract(.{
            .function_name = "NewLocalRef",
            .contract_type = .gc_managed,
            .ownership = .callee_owns,  // 函数返回时自动释放
            .error_prone = false,
            .source = .jni,
        });
        
        try self.addContract(.{
            .function_name = "NewStringUTF",
            .contract_type = .gc_managed,
            .ownership = .callee_owns,
            .source = .jni,
        });
        
        // ========== POSIX ==========
        try self.addContract(.{
            .function_name = "open",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"close"},
            .ownership = .caller_owns,
            .source = .posix,
        });
        
        try self.addContract(.{
            .function_name = "socket",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"close"},
            .ownership = .caller_owns,
            .source = .posix,
        });
        
        try self.addContract(.{
            .function_name = "mmap",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"munmap"},
            .ownership = .caller_owns,
            .source = .posix,
        });
        
        // ========== GLib ==========
        try self.addContract(.{
            .function_name = "g_malloc",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"g_free"},
            .ownership = .caller_owns,
            .source = .glib,
        });
        
        try self.addContract(.{
            .function_name = "g_strdup",
            .contract_type = .allocator,
            .paired_release = &[_][]const u8{"g_free"},
            .ownership = .caller_owns,
            .source = .glib,
        });
    }
    
    pub fn addContract(self: *FFIContractDB, contract: FFIContract) !void {
        try self.contracts.put(contract.function_name, contract);
    }
    
    /// 查询函数的契约
    pub fn getContract(self: *const FFIContractDB, func_name: []const u8) ?*const FFIContract {
        return self.contracts.getPtr(func_name);
    }
    
    /// 检查释放函数是否与分配函数配对
    pub fn isValidPair(
        self: *const FFIContractDB,
        alloc_func: []const u8,
        free_func: []const u8,
    ) ?PairValidity {
        const alloc_contract = self.getContract(alloc_func) orelse return null;
        
        if (alloc_contract.paired_release) |valid_releases| {
            for (valid_releases) |valid_free| {
                if (std.mem.eql(u8, free_func, valid_free)) {
                    return .valid;
                }
            }
            return .mismatch;  // 错误的配对! 这是潜在bug
        }
        
        return null;  // 未知
    }
    
    /// 检查是否应该报告泄漏
    pub fn shouldReportLeak(
        self: *const FFIContractDB,
        alloc_func: []const u8,
    ) bool {
        if (self.getContract(alloc_func)) |contract| {
            // GC管理的对象不报告泄漏
            if (contract.contract_type == .gc_managed) return false;
            
            // 借用引用不报告泄漏
            if (contract.ownership == .borrowed) return false;
            
            // 被调用者拥有的不报告泄漏
            if (contract.ownership == .callee_owns) return false;
            
            return true;
        }
        
        // 未知函数，保守地报告
        return true;
    }
    
    pub const PairValidity = enum {
        valid,      // 正确配对
        mismatch,   // 错误配对 (潜在bug!)
    };
};
```

```zig
// 在 validateFreeWithMemoryGraph 函数中添加契约检查
pub fn validateFreeWithMemoryGraph(
    // ... 现有参数 ...
    contract_db: *const FFIContractDB,  // 新增参数
) !void {
    // ... 现有逻辑 ...
    
    // 新增: 检查配对有效性
    if (alloc_func_name) |alloc_name| {
        if (free_func_name) |free_name| {
            if (contract_db.isValidPair(alloc_name, free_name)) |validity| {
                switch (validity) {
                    .valid => {
                        // 正确配对，继续
                    },
                    .mismatch => {
                        // 错误配对! 报告为 cross_language_free_mismatch
                        try reportMismatch(alloc_name, free_name);
                        return;
                    },
                }
            }
        }
    }
}
```
