```zig
pub fn trackRustDrop(graph: *MemoryGraph, inst: c.LLVMValueRef) !void {
    const called = c.LLVMGetCalledValue(inst);
    const name = std.mem.span(c.LLVMGetValueName(called));
    
    if (std.mem.indexOf(u8, name, "drop_in_place") != null or
        std.mem.indexOf(u8, name, "__rust_dealloc") != null)
    {
        const ptr_arg = c.LLVMGetOperand(inst, 0);
        const ptr_val = @intFromPtr(ptr_arg);
        
        if (graph.findCanonicalAlloc(ptr_val)) |node| {
            node.has_raii_cleanup = true;
            try node.raii_cleanup_sites.append(@intFromPtr(inst));
        }
    }
}
```

**新增 Pass**: `src/pass/rust_ownership.zig`

```zig
pub const RustOwnershipPass = struct {
    pub fn run(ctx: *PassContext) !void {
        for (ctx.module_functions) |func| {
            var bb = c.LLVMGetFirstBasicBlock(func);
            while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
                var inst = c.LLVMGetFirstInstruction(bb);
                while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                    if (c.LLVMIsACallInst(inst) != null) {
                        try ctx.memory_graph.trackRustDrop(inst);
                    }
                }
            }
        }
    }
};
```

```zig
        if (std.mem.indexOf(u8, func_name, "Box") != null) return .rust_box;

        if (std.mem.indexOf(u8, func_name, "String") != null) return .rust_string;
        if (std.mem.indexOf(u8, func_name, "Rc") != null) return .rust_rc;
```

```zig
pub fn validateFreeWithMemoryGraph(
    // ... 现有参数 ...
    contract_db: *const FFIContractDB,  // 新增

    // 检查配对有效性
    if (alloc_func_name) |alloc_name| {

            if (contract_db.isValidPair(alloc_name, free_name)) |validity| {
                switch (validity) {
```

#### Day 18-19: 跨函数生命周期追踪
**文件**: `src/semantics/memory_graph.zig` (增强)

```zig
pub fn isEventuallyFreed(
    graph: *MemoryGraph,
    ptr_val: u64,
    max_depth: u32,
) bool {
    // 递归追踪调用链
    if (max_depth == 0) return false;
    
    // 检查直接释放
    if (graph.findCanonicalAlloc(ptr_val)) |node| {
        if (node.freed) return true;
        if (node.has_raii_cleanup) return true;
        if (node.is_gc_managed) return true;
    }
    
    // 检查返回给调用者后是否被释放
    const ret_edges = graph.getCallRetsForPtr(ptr_val);
    for (ret_edges) |ret_edge| {
        if (graph.isEventuallyFreed(ret_edge.ret_ptr, max_depth - 1)) {
            return true;
        }
    }
    
    return false;
}
```
