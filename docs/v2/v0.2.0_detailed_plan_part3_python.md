# OmniScope 0.2.0 详细开发方案 - Part 3: Python适配器

## 🐍 Python Adapter 完整实现

### 文件位置: `src/lang/python_adapter.zig` (新建)

---

## 核心功能

1. **Python C API 函数分类** - 识别新引用/借用引用/窃取引用
2. **引用计数追踪** - 追踪 Py_INCREF/Py_DECREF 平衡
3. **GIL 操作检测** - 识别 PyGILState_Ensure/Release
4. **异常路径分析** - 检测 PyErr_SetString 后的资源泄漏

## 完整代码实现

```zig
const std = @import("std");
const c = @import("../ir/llvm_raw.zig").c;
const SemanticKind = @import("../semantics/semantic_tree.zig").SemanticKind;
const MemoryGraph = @import("../semantics/memory_graph.zig").MemoryGraph;
const log = @import("../common/log.zig");

pub const PythonAdapter = struct {
    allocator: std.mem.Allocator,
    
    /// Python C API 函数分类
    const PYTHON_CAPI = struct {
        // 返回新引用 (owned reference) - 调用者必须 Py_DECREF
        const NEW_REF_FUNCTIONS = [_][]const u8{
            "PyBytes_FromString",
            "PyBytes_FromStringAndSize",
            "PyUnicode_FromString",
            "PyUnicode_FromFormat",
            "PyList_New",
            "PyDict_New",
            "PyTuple_New",
            "PySet_New",
            "PyObject_CallObject",
            "PyObject_CallFunction",
            "PyObject_CallMethod",
            "Py_BuildValue",
            "PyImport_ImportModule",
            "PyImport_Import",
            "PyObject_GetAttr",
            "PyObject_GetAttrString",
            "PyObject_GetItem",
            "PyDict_GetItemString",  // 注意: 这个实际是借用引用
            "PyErr_Format",
        };
        
        // 返回借用引用 (borrowed reference) - 不应 Py_DECREF
        const BORROWED_REF_FUNCTIONS = [_][]const u8{
            "PyList_GetItem",
            "PyTuple_GetItem",
            "PyDict_GetItem",
            "PyDict_GetItemString",
            "PyImport_AddModule",
            "PyModule_GetDict",
            "PyErr_Occurred",
            "PySys_GetObject",
            "PyImport_GetModuleDict",
        };
        
        // 窃取引用 (steals reference) - 函数接管所有权
        const STEAL_REF_FUNCTIONS = [_][]const u8{
            "PyList_SetItem",
            "PyTuple_SetItem",
            "PyModule_AddObject",
            "PyDict_SetItem",  // 不窃取，但需要注意
        };
        
        // GIL 操作
        const GIL_FUNCTIONS = [_][]const u8{
            "PyGILState_Ensure",
            "PyGILState_Release",
            "PyEval_SaveThread",
            "PyEval_RestoreThread",
            "Py_BEGIN_ALLOW_THREADS",
            "Py_END_ALLOW_THREADS",
        };
        
        // 异常处理
        const EXCEPTION_FUNCTIONS = [_][]const u8{
            "PyErr_SetString",
            "PyErr_Format",
            "PyErr_Occurred",
            "PyErr_Clear",
            "PyErr_Print",
        };
    };
    
    pub fn init(allocator: std.mem.Allocator) PythonAdapter {
        return .{ .allocator = allocator };
    }
    
    /// 分类 Python C API 调用
    pub fn classifyCall(func_name: []const u8) ?PythonCallSemantics {
        // 检查是否是新引用函数
        for (PYTHON_CAPI.NEW_REF_FUNCTIONS) |f| {
            if (std.mem.eql(u8, func_name, f)) {
                return .{ 
                    .kind = .new_reference, 
                    .ownership = .caller_owns,
                    .semantic_kind = .python_owned_ref,
                };
            }
        }
        
        // 检查是否是借用引用函数
        for (PYTHON_CAPI.BORROWED_REF_FUNCTIONS) |f| {
            if (std.mem.eql(u8, func_name, f)) {
                return .{ 
                    .kind = .borrowed_reference, 
                    .ownership = .borrowed,
                    .semantic_kind = .python_borrowed_ref,
                };
            }
        }
        
        // 检查是否是窃取引用函数
        for (PYTHON_CAPI.STEAL_REF_FUNCTIONS) |f| {
            if (std.mem.eql(u8, func_name, f)) {
                return .{ 
                    .kind = .steal_reference, 
                    .ownership = .transferred,
                    .semantic_kind = .python_owned_ref,
                };
            }
        }
        
        return null;
    }
    
    /// 检测 Python 引用计数操作
    pub fn detectRefCountOp(inst: c.LLVMValueRef) ?RefCountOp {
        if (c.LLVMIsACallInst(inst) == null) return null;
        
        const callee = c.LLVMGetCalledValue(inst);
        const name = std.mem.span(c.LLVMGetValueName(callee));
        
        if (std.mem.indexOf(u8, name, "Py_INCREF") != null) {
            return .{ 
                .op_type = .increment, 
                .ptr_arg = c.LLVMGetOperand(inst, 0),
                .semantic_kind = .python_refcount_inc,
            };
        }
        
        if (std.mem.indexOf(u8, name, "Py_DECREF") != null) {
            return .{ 
                .op_type = .decrement, 
                .ptr_arg = c.LLVMGetOperand(inst, 0),
                .semantic_kind = .python_refcount_dec,
            };
        }
        
        if (std.mem.indexOf(u8, name, "Py_XINCREF") != null) {
            return .{ 
                .op_type = .increment_nullable, 
                .ptr_arg = c.LLVMGetOperand(inst, 0),
                .semantic_kind = .python_refcount_inc,
            };
        }
        
        if (std.mem.indexOf(u8, name, "Py_XDECREF") != null) {
            return .{ 
                .op_type = .decrement_nullable, 
                .ptr_arg = c.LLVMGetOperand(inst, 0),
                .semantic_kind = .python_refcount_dec,
            };
        }
        
        return null;
    }
    
    /// 追踪 Python 对象的引用计数平衡
    pub fn trackRefCountBalance(
        self: *PythonAdapter,
        graph: *MemoryGraph,
        func: c.LLVMValueRef,
    ) !RefCountBalance {
        var balance = RefCountBalance{};
        var ptr_balances = std.AutoHashMap(u64, i32).init(self.allocator);
        defer ptr_balances.deinit();
        
        // 遍历函数中的所有指令
        var bb = c.LLVMGetFirstBasicBlock(func);
        while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
            var inst = c.LLVMGetFirstInstruction(bb);
            while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                if (detectRefCountOp(inst)) |op| {
                    const ptr_val = @intFromPtr(op.ptr_arg);
                    
                    switch (op.op_type) {
                        .increment, .increment_nullable => {
                            balance.increments += 1;
                            const current = ptr_balances.get(ptr_val) orelse 0;
                            try ptr_balances.put(ptr_val, current + 1);
                        },
                        .decrement, .decrement_nullable => {
                            balance.decrements += 1;
                            const current = ptr_balances.get(ptr_val) orelse 0;
                            try ptr_balances.put(ptr_val, current - 1);
                        },
                    }
                    
                    // 更新 MemoryGraph 中的引用计数信息
                    if (graph.findCanonicalAlloc(ptr_val)) |node| {
                        node.refcount_ops.increments = balance.increments;
                        node.refcount_ops.decrements = balance.decrements;
                        node.ownership_model = .refcount;
                    }
                }
            }
        }
        
        // 检查每个指针的平衡
        var iter = ptr_balances.iterator();
        while (iter.next()) |entry| {
            if (entry.value_ptr.* != 0) {
                balance.unbalanced_ptrs += 1;
                log.warn("Python refcount imbalance: ptr=0x{x} net={d}", .{
                    entry.key_ptr.*, entry.value_ptr.*
                });
            }
        }
        
        return balance;
    }
    
    /// 检测 GIL 操作
    pub fn detectGILOp(inst: c.LLVMValueRef) ?GILOp {
        if (c.LLVMIsACallInst(inst) == null) return null;
        
        const callee = c.LLVMGetCalledValue(inst);
        const name = std.mem.span(c.LLVMGetValueName(callee));
        
        if (std.mem.eql(u8, name, "PyGILState_Ensure")) {
            return .{ .op_type = .acquire };
        }
        
        if (std.mem.eql(u8, name, "PyGILState_Release")) {
            return .{ .op_type = .release };
        }
        
        return null;
    }
    
    /// 检测异常路径上的资源泄漏
    pub fn detectExceptionLeak(
        self: *PythonAdapter,
        graph: *MemoryGraph,
        func: c.LLVMValueRef,
    ) !std.ArrayList(ExceptionLeak) {
        var leaks = std.ArrayList(ExceptionLeak).init(self.allocator);
        
        // 查找所有 PyErr_SetString 调用
        var bb = c.LLVMGetFirstBasicBlock(func);
        while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
            var inst = c.LLVMGetFirstInstruction(bb);
            while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                if (c.LLVMIsACallInst(inst) != null) {
                    const callee = c.LLVMGetCalledValue(inst);
                    const name = std.mem.span(c.LLVMGetValueName(callee));
                    
                    if (std.mem.eql(u8, name, "PyErr_SetString") or
                        std.mem.eql(u8, name, "PyErr_Format")) {
                        // 检查此基本块之前是否有未释放的分配
                        // TODO: 实现详细的异常路径分析
                    }
                }
            }
        }
        
        return leaks;
    }
};

pub const PythonCallSemantics = struct {
    kind: enum {
        new_reference,      // 返回新引用
        borrowed_reference, // 返回借用引用
        steal_reference,    // 窃取引用
    },
    ownership: enum {
        caller_owns,   // 调用者拥有
        borrowed,      // 借用
        transferred,   // 转移给被调用者
    },
    semantic_kind: SemanticKind,
};

pub const RefCountOp = struct {
    op_type: enum {
        increment,
        decrement,
        increment_nullable,  // Py_XINCREF
        decrement_nullable,  // Py_XDECREF
    },
    ptr_arg: c.LLVMValueRef,
    semantic_kind: SemanticKind,
};

pub const RefCountBalance = struct {
    increments: u32 = 0,
    decrements: u32 = 0,
    unbalanced_ptrs: u32 = 0,
    
    pub fn isBalanced(self: RefCountBalance) bool {
        return self.increments == self.decrements and self.unbalanced_ptrs == 0;
    }
    
    pub fn netChange(self: RefCountBalance) i32 {
        return @as(i32, @intCast(self.increments)) - 
               @as(i32, @intCast(self.decrements));
    }
};

pub const GILOp = struct {
    op_type: enum {
        acquire,  // PyGILState_Ensure
        release,  // PyGILState_Release
    },
};

pub const ExceptionLeak = struct {
    alloc_site: u64,
    exception_site: u64,
    leaked_ptr: u64,
};
```

## 使用示例

```zig
// 在 PassManager 中集成 Python Adapter
const python_adapter = try PythonAdapter.init(allocator);

// 分析函数
const balance = try python_adapter.trackRefCountBalance(&memory_graph, func);

if (!balance.isBalanced()) {
    log.warn("Python refcount imbalance detected: +{d} -{d}", .{
        balance.increments, balance.decrements
    });
}
```
