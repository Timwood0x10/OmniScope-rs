## C++ 内存管理特点

1. **RAII (Resource Acquisition Is Initialization)** - 作用域结束自动释放
2. **智能指针** - unique_ptr, shared_ptr, weak_ptr
3. **STL 容器** - vector, string, map 自动管理内存
4. **异常安全** - try/catch 块中的资源泄漏

## 完整代码实现

```zig
const std = @import("std");
const c = @import("../ir/llvm_raw.zig").c;
const SemanticKind = @import("../semantics/semantic_tree.zig").SemanticKind;
const MemoryGraph = @import("../semantics/memory_graph.zig").MemoryGraph;
const log = @import("../common/log.zig");

pub const CppAdapter = struct {
    allocator: std.mem.Allocator,
    
    /// C++ 智能指针操作
    const SMART_PTR_OPS = struct {
        const UNIQUE_PTR = [_][]const u8{
            "std::unique_ptr",
            "std::__1::unique_ptr",  // libc++
            "std::__cxx11::unique_ptr",  // libstdc++
            "_ZNSt10unique_ptr",  // mangled
        };
        
        const SHARED_PTR = [_][]const u8{
            "std::shared_ptr",
            "std::__1::shared_ptr",
            "std::__cxx11::shared_ptr",
            "_ZNSt10shared_ptr",
        };
        
        const WEAK_PTR = [_][]const u8{
            "std::weak_ptr",
            "std::__1::weak_ptr",
        };
    };
    
    /// STL 容器构造/析构
    const STL_CONTAINERS = struct {
        const VECTOR = [_][]const u8{
            "std::vector",
            "std::__1::vector",
            "_ZNSt6vector",
        };
        
        const STRING = [_][]const u8{
            "std::string",
            "std::basic_string",
            "std::__cxx11::basic_string",
            "_ZNSt7__cxx1112basic_string",
        };
        
        const MAP = [_][]const u8{
            "std::map",
            "std::unordered_map",
            "_ZNSt3map",
        };
    };
    
    pub fn init(allocator: std.mem.Allocator) CppAdapter {
        return .{ .allocator = allocator };
    }
    
    /// 检测 C++ 析构函数
    pub fn isDestructor(func_name: []const u8) bool {
        // C++ 析构函数的 mangled 名称包含 "D0" 或 "D1" 或 "D2"
        // 例如: _ZN7MyClassD1Ev (MyClass::~MyClass())
        if (std.mem.indexOf(u8, func_name, "D0Ev") != null or
            std.mem.indexOf(u8, func_name, "D1Ev") != null or
            std.mem.indexOf(u8, func_name, "D2Ev") != null) {
            return true;
        }
        
        return false;
    }
    
    /// 检测智能指针类型
    pub fn detectSmartPtr(type_name: []const u8) ?SmartPtrType {
        // unique_ptr
        for (SMART_PTR_OPS.UNIQUE_PTR) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return .{ .kind = .unique_ptr, .semantic_kind = .cpp_unique_ptr };
            }
        }
        
        // shared_ptr
        for (SMART_PTR_OPS.SHARED_PTR) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return .{ .kind = .shared_ptr, .semantic_kind = .cpp_shared_ptr };
            }
        }
        
        // weak_ptr
        for (SMART_PTR_OPS.WEAK_PTR) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return .{ .kind = .weak_ptr, .semantic_kind = .cpp_shared_ptr };
            }
        }
        
        return null;
    }
    
    /// 检测 STL 容器
    pub fn detectSTLContainer(type_name: []const u8) ?STLContainerType {
        // vector
        for (STL_CONTAINERS.VECTOR) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return .{ .kind = .vector, .manages_memory = true };
            }
        }
        
        // string
        for (STL_CONTAINERS.STRING) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return .{ .kind = .string, .manages_memory = true };
            }
        }
        
        // map
        for (STL_CONTAINERS.MAP) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return .{ .kind = .map, .manages_memory = true };
            }
        }
        
        return null;
    }
    
    /// 查找析构函数调用
    pub fn findDestructorCalls(
        self: *CppAdapter,
        func: c.LLVMValueRef,
    ) !std.ArrayList(DestructorCall) {
        var destructor_calls = std.ArrayList(DestructorCall).init(self.allocator);
        
        var bb = c.LLVMGetFirstBasicBlock(func);
        while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
            var inst = c.LLVMGetFirstInstruction(bb);
            while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                if (c.LLVMIsACallInst(inst) != null) {
                    const callee = c.LLVMGetCalledValue(inst);
                    const name = std.mem.span(c.LLVMGetValueName(callee));
                    
                    if (isDestructor(name)) {
                        try destructor_calls.append(.{
                            .call_site = @intFromPtr(inst),
                            .destructor_name = name,
                            .object_ptr = @intFromPtr(c.LLVMGetOperand(inst, 0)),
                            .semantic_kind = .cpp_destructor,
                        });
                    }
                }
            }
        }
        
        return destructor_calls;
    }
    
    /// 检测异常路径上的资源泄漏
    pub fn detectExceptionLeak(
        self: *CppAdapter,
        graph: *MemoryGraph,
        func: c.LLVMValueRef,
    ) !std.ArrayList(ExceptionLeak) {
        var leaks = std.ArrayList(ExceptionLeak).init(self.allocator);
        
        // 查找所有 invoke 指令 (C++ 异常处理)
        var bb = c.LLVMGetFirstBasicBlock(func);
        while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
            var inst = c.LLVMGetFirstInstruction(bb);
            while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                if (c.LLVMIsAInvokeInst(inst) != null) {
                    // invoke 指令有两个后继: normal 和 unwind
                    const normal_dest = c.LLVMGetNormalDest(inst);
                    const unwind_dest = c.LLVMGetUnwindDest(inst);
                    
                    // 检查 unwind 路径是否有资源清理
                    _ = normal_dest;
                    _ = unwind_dest;
                    _ = graph;
                    
                    // TODO: 实现完整的异常路径分析
                }
            }
        }
        
        return leaks;
    }
    
    /// 检测 unique_ptr::release() 调用 (可能导致泄漏)
    pub fn detectUniquePtrRelease(
        self: *CppAdapter,
        func: c.LLVMValueRef,
    ) !std.ArrayList(UniquePtrRelease) {
        var releases = std.ArrayList(UniquePtrRelease).init(self.allocator);
        
        var bb = c.LLVMGetFirstBasicBlock(func);
        while (bb != null) : (bb = c.LLVMGetNextBasicBlock(bb)) {
            var inst = c.LLVMGetFirstInstruction(bb);
            while (inst != null) : (inst = c.LLVMGetNextInstruction(inst)) {
                if (c.LLVMIsACallInst(inst) != null) {
                    const callee = c.LLVMGetCalledValue(inst);
                    const name = std.mem.span(c.LLVMGetValueName(callee));
                    
                    // 检测 unique_ptr::release()
                    if (std.mem.indexOf(u8, name, "unique_ptr") != null and
                        std.mem.indexOf(u8, name, "release") != null) {
                        try releases.append(.{
                            .call_site = @intFromPtr(inst),
                            .released_ptr = @intFromPtr(c.LLVMGetOperand(inst, 0)),
                            .warning = "unique_ptr::release() transfers ownership - ensure manual delete",
                        });
                    }
                }
            }
        }
        
        return releases;
    }
};

pub const SmartPtrType = struct {
    kind: enum {
        unique_ptr,
        shared_ptr,
        weak_ptr,
    },
    semantic_kind: SemanticKind,
};

pub const STLContainerType = struct {
    kind: enum {
        vector,
        string,
        map,
        set,
        list,
    },
    manages_memory: bool,
};

pub const DestructorCall = struct {
    call_site: u64,
    destructor_name: []const u8,
    object_ptr: u64,
    semantic_kind: SemanticKind,
};

pub const ExceptionLeak = struct {
    alloc_site: u64,
    invoke_site: u64,
    unwind_dest: u64,
    leaked_ptr: u64,
};

pub const UniquePtrRelease = struct {
    call_site: u64,
    released_ptr: u64,
    warning: []const u8,
};
```

```zig
const std = @import("std");
const c = @import("../ir/llvm_raw.zig").c;
const SemanticKind = @import("../semantics/semantic_tree.zig").SemanticKind;

pub const CSharpAdapter = struct {
    allocator: std.mem.Allocator,
    
    /// C# P/Invoke 函数模式
    const PINVOKE_PATTERNS = [_][]const u8{
        "PInvoke",
        "DllImport",
        "Marshal",
    };
    
    /// SafeHandle 模式
    const SAFE_HANDLE_PATTERNS = [_][]const u8{
        "SafeHandle",
        "CriticalHandle",
        "ReleaseHandle",
    };
    
    pub fn init(allocator: std.mem.Allocator) CSharpAdapter {
        return .{ .allocator = allocator };
    }
    
    /// 检测 SafeHandle 模式
    pub fn isSafeHandle(type_name: []const u8) bool {
        for (SAFE_HANDLE_PATTERNS) |pattern| {
            if (std.mem.indexOf(u8, type_name, pattern) != null) {
                return true;
            }
        }
        return false;
    }
    
    /// 检测 P/Invoke 调用
    pub fn isPInvokeCall(func_name: []const u8) bool {
        for (PINVOKE_PATTERNS) |pattern| {
            if (std.mem.indexOf(u8, func_name, pattern) != null) {
                return true;
            }
        }
        return false;
    }
    
    /// 检测 IDisposable.Dispose 调用
    pub fn isDisposeCall(func_name: []const u8) bool {
        return std.mem.indexOf(u8, func_name, "Dispose") != null;
    }
};
```

## ☕ Java/JNI Adapter 实现

### 文件位置: `src/lang/java_adapter.zig` (新建)

```zig
const std = @import("std");
const c = @import("../ir/llvm_raw.zig").c;
const SemanticKind = @import("../semantics/semantic_tree.zig").SemanticKind;

pub const JavaAdapter = struct {
    allocator: std.mem.Allocator,
    
    /// JNI 引用类型函数
    const JNI_FUNCTIONS = struct {
        const LOCAL_REF = [_][]const u8{
            "NewLocalRef",
            "NewStringUTF",
            "NewObject",
            "NewObjectArray",
        };
        
        const GLOBAL_REF = [_][]const u8{
            "NewGlobalRef",
            "DeleteGlobalRef",
        };
        
        const WEAK_REF = [_][]const u8{
            "NewWeakGlobalRef",
            "DeleteWeakGlobalRef",
        };
    };
    
    pub fn init(allocator: std.mem.Allocator) JavaAdapter {
        return .{ .allocator = allocator };
    }
    
    /// 分类 JNI 引用
    pub fn classifyJNIRef(func_name: []const u8) ?JNIRefType {
        // Local reference
        for (JNI_FUNCTIONS.LOCAL_REF) |f| {
            if (std.mem.eql(u8, func_name, f)) {
                return .{ 
                    .kind = .local_ref, 
                    .auto_freed = true,
                    .semantic_kind = .java_local_ref,
                };
            }
        }
        
        // Global reference
        for (JNI_FUNCTIONS.GLOBAL_REF) |f| {
            if (std.mem.eql(u8, func_name, f)) {
                return .{ 
                    .kind = .global_ref, 
                    .auto_freed = false,
                    .semantic_kind = .java_global_ref,
                };
            }
        }
        
        // Weak reference
        for (JNI_FUNCTIONS.WEAK_REF) |f| {
            if (std.mem.eql(u8, func_name, f)) {
                return .{ 
                    .kind = .weak_ref, 
                    .auto_freed = true,
                    .semantic_kind = .java_weak_ref,
                };
            }
        }
        
        return null;
    }
};

pub const JNIRefType = struct {
    kind: enum {
        local_ref,   // 函数返回时自动释放
        global_ref,  // 需要手动 DeleteGlobalRef
        weak_ref,    // 可能被 GC 回收
    },
    auto_freed: bool,
    semantic_kind: SemanticKind,
};
```
